>> For Loop:
for (int counter = 0; counter < 100; counter++) {
    // do something
}


>> While Loop:
int i = 0;
while (i < 11) {
    // do something
    i++;
}


>> Do-While Loop:
do {
    // do something
} while (condition);


>> Nested Loops:
for (...) {
    for (...) {
        // nested loop logic
    }
}
